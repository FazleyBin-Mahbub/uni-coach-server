

const express = require("express");
const cors = require("cors");
const { MongoClient } = require("mongodb");
require("dotenv").config();
const app = express();
const port = process.env.PORT || 9000;

// middleware
app.use(cors());
app.use(express.json());
// username: ariyan34
// password: mongodbariyan
const uri = `mongodb+srv://${process.env.DB_USER}:${process.env.DB_PASS}@cluster0.58ekm.mongodb.net/myFirstDatabase?retryWrites=true&w=majority`;

const client = new MongoClient(uri, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
});

async function run() {
  try {
    await client.connect();
    const database = client.db("uni-coach");
    const servicesCollection = database.collection("services");
    // POST API request
    app.post("/services", async (req, res) => {
      const service = req.body;
      console.log("hit the post", service);
      const result = await servicesCollection.insertOne(service);
      console.log(result);
      res.send(result);
    });
  } finally {
  }
}
run().catch(console.dir);
app.get("/", (req, res) => {
  res.send("running uni-coach-server");
});

app.listen(port, (req, res) => {
  console.log("running the port: " + port);
});
